    # 3.演示矩阵乘法操作,对应元素直接做 相应的操作

    # print(f"t3:{t3}")

# 3.测试
if __name__ == '__main__':

#### 梯度下降法求最优解

梯度形式是不变的,梯度的值是变的(2w)



![image-20260317205405512](深度学习-img/image-20260317205405512.png)



![image-20260317213827587](深度学习-img/image-20260317213827587.png)

```py
"""
演示自动微分模块,循环实现 计算梯度,更新参数

需求:
    求 y = x**2 + 20 的极小值点 并打印y是最小值时 w的值(梯度)

解题步骤:
# 1. 定义点 w=10 requires_grad=True  dtpe=torc
# 2. 定义函数 loss = w**2 + 20
# 3. 利用梯度下降法 循环迭代1000 求最优解
# 3.1 正向计算(前向传播)
# 3.2 梯度清零 w.grad.zero_()
# 3.3 反向传播
# 3.4 梯度更新 w.data = w.data - 0.01 * w.grad

回顾
    W新=W旧-学习率 * 梯度
"""
import torch

# 1. 定义点 w=10 requires_grad=True  dtpe=torch.float32
# 参1:初始的权重 w(旧),参2:自动微分 (求导),参数3:数据类型 浮点型
w=torch.tensor(10,requires_grad= True,dtype=torch.float32)

# 2. 定义函数 loss = w**2 + 20
loss=w**2 + 20          # 求导:loss'=2w


# 3. 利用梯度下降法 循环迭代1000 求最优解
print(f"开始 权重初始值:{w},(0.01*w.grad):无,loss:{loss}")

# 迭代100次,求最优解   [其实执行了101次]
for i in range(1,101):
    # 3.1 正向计算(前向传播)
    loss=w**2 + 20

    # 3.2 梯度清零 w.grad.zero_()   默认梯度会累加
    # 至此(第一次的时候),还没有计算梯度,所以 w.grad=None,要做非空判断
    if w.grad is not None:
        w.grad.zero_()

    # 3.3 反向传播
    loss.sum().backward()

    # 3.4 梯度更新 w.data = w.data - 0.01 * w.grad        因为梯度是2w，这里的w.grad就是把 w 带进去了得到梯度值
    # print(f"梯度值为:{w.grad}")
    w.data = w.data - 0.01 * w.grad

    # 3.5 打印本次 梯度更新后 权重参数结果
    print(f"第{i}次 权重初始值:{w},(0.01*w.grad):{0.01*w.grad:.5f},loss:{loss:.5f}")


# 4.打印最终结果
print(f"最终结果   权重:{w},梯度:{w.grad:.5f},loss:{loss:.5f}")
```

> ### 详细解释
>
> 1. **nn.CrossEntropyLoss 的本质**
>
>    PyTorch 的 `nn.CrossEntropyLoss()` 是一个**组合函数**，它内部已经包含了两个步骤：
>
>    
>
>    - Step 1: 对输入的预测值（logits，原始得分）做 `Softmax` 归一化（转成概率）
>    - Step 2: 计算 `log_softmax` 后的值与真实标签的负对数似然损失（NLLLoss）
>
>    
>
>    公式等价于：`CrossEntropyLoss = log_softmax + NLLLoss`

> 这个初始梯度1是根据损失函数的导函数算的，当w为1时，带入损失函数一阶导2w/2得当前梯度为1,也就是当前梯度

> AI：第一次更新权重时，动量项还没有历史梯度可以累积，所以它用的就是当前这一步的梯度，而不是 “根据动量计算的梯度”。

> 都讲了这个办法是将梯度带入指数加权公式中的Yt了
>
> 所以视频这里没讲错

![image-20260323215433801](深度学习-img/image-20260323215433801.png)

> PyTorch 中，张量的梯度（`.grad`）不会自动清零，而是**每次调用`.backward()`时，将当前计算的梯度累加到已有梯度上**（即 `w.grad += 本次梯度`）。
>
> w是权重,w.grad是损失函数对权重的导数(权重的梯度)

```py
    import torch
    import torch.optim as optim

    # 完全复刻你代码中的权重初始化（保持一致）
    w = torch.tensor(1.0, requires_grad=True, dtype=torch.float)
    print(f"初始状态: w={w.item()}, w.grad={w.grad}")  # 初始梯度为None（未计算）

    # 第一次计算梯度（不做梯度清零）
    criterion1 = (w ** 2) / 2.0  # 和你代码中损失函数一致
    criterion1.sum().backward()  # 第一次反向传播，计算梯度
    print(f"第一次backward()后: w={w.item()}, w.grad={w.grad.item()}")  # 梯度=1.0（w的导数是w）

    # 第二次计算梯度（不做梯度清零，关键验证累积）
    criterion2 = (w ** 2) / 2.0  # 基于当前w（仍为1.0）重新计算损失
    criterion2.sum().backward()  # 第二次反向传播，梯度会累加到已有值上
    print(f"第二次backward()后: w={w.item()}, w.grad={w.grad.item()}")  # 梯度=1.0+1.0=2.0

    # 第三次计算梯度（继续累积）
    criterion3 = (w ** 2) / 2.0
    criterion3.sum().backward()
    print(f"第三次backward()后: w={w.item()}, w.grad={w.grad.item()}")  # 梯度=2.0+1.0=3.0

    # 对比：执行梯度清零后，梯度不再累积
    optimizer = optim.SGD(params=[w], lr=0.01, momentum=0.9)
    optimizer.zero_grad()  # 手动梯度清零
    criterion4 = (w ** 2) / 2.0
    criterion4.sum().backward()
    print(f"梯度清零后，第四次backward(): w.grad={w.grad.item()}")  # 梯度重置为1.0，不再累积
```



#### 梯度下降的优化方法-adagrad

> 点乘:各个分量相乘

![image-20260323221528595](深度学习-img/image-20260323221528595.png)

> 学习率包是递减的啊，一开始大是因为还没有找到最小值，等后面快要到最小值的时候学习率变小点不就能找到了吗，要不然你就梯度振荡了



```py
def dm02_adagrad():
    # 1.初始化权重参数
    w = torch.tensor(1.0, requires_grad=True, dtype=torch.float)
    # 2.定义损失函数
    criterion = ((w ** 2) / 2.0)
    # 3.创建优化器(函数对象)
    # 思路1:基于SGD(随机梯度下降),加入参数 momentum,就是动量法
    # 参1:(待优化的)参数列表,参2:学习率,参3:动量参数
    # optimizer = optim.SGD(params=[w], lr=0.01, momentum=0.9)  # 细节:(β调节权重系数)momentum=0(默认),只考虑:本次梯度

    # 思路2:基于AdaGrad(自适应学习率).
    optimizer=optim.Adagrad(params=[w],lr=0.01)

    # 4.计算梯度值:梯度清零 + 反向传播 + 参数更新
    optimizer.zero_grad()
    criterion.sum().backward()
    optimizer.step()
    print(f"w:{w},w_grad:{w.grad}")

    # 5. 重复上述的步骤,第2次 更新权重参数
    # 定义损失函数
    criterion = ((w ** 2) / 2.0)
    # 计算梯度值: 梯度清零 + 反向传播 + 参数更新
    optimizer.zero_grad()
    criterion.sum().backward()
    optimizer.step()
    # 打印结果
    print(f"w:{w},w_grad:{w.grad}")
```

### 损失函数求导得到梯度，损失函数具体到底是对谁求导？？

损失函数求导（计算梯度）的核心目标是：**对模型中所有可学习的参数（权重 / 偏置）求导**，最终目的是通过梯度下降更新这些参数，让模型预测更接近真实值。

**模型参数**：神经网络中需要学习的变量，比如全连接层的权重 `W`、偏置 `b`，卷积层的卷积核等（PyTorch 中这些参数会被标记为 `requires_grad=True`）。

**求导本质**：损失函数是模型参数的 “函数”（因为预测值由参数计算而来），求导就是计算 “损失对每个参数的变化率”—— 参数变一点点，损失会变多少。

![image-20260317214619178](深度学习-img/image-20260317214619178.png)



### 梯度形式是不变的，梯度值是变的

```py
# 2. 定义函数 loss = w**2 + 20
loss=w**2 + 20          # 求导:loss'=2w
```

传播过程中：

梯度（导数结果式子）形式永远是 2w

但是梯度值是变的（权重 w 一直在更新）
